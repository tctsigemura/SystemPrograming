# フォアグラウンド・ジョブの場合
$ sleep 1000                            <-- ジョブの開始
^C                                      <-- 実行を終了させる
$                                       <-- 次のコマンド入力が可能になる

# バックグラウンド・ジョブの場合
$  sleep 1000 &                         <-- & 付きで開始
[1] 6123                                <-- ジョブ番号=1，PID=6123
$                                       <-- すぐに次のコマンド入力が可能
