# 通常のコマンド実行
% vi hello.c                                     <-- 1プロセスが1ジョブ

# パイプを使用しファイルサイズ順にソートして表示
% ls -l | sort -n --key=5                        <-- 2プロセスが1ジョブ

# 二つのコマンド（ジョブ）を順次実行
% touch a.txt; chmod 777 a.txt                   <-- 2ジョブ

# 二つのコマンド（ジョブ）を並列実行
% touch a.txt & touch b.txt                      <-- 2ジョブ
